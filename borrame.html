Cómo funciona

El proceso para deshabilitar botones durante peticiones incluye:

  1. Al iniciar la petición: Deshabilitar el botón y mostrar un indicador de carga
  2. Durante la petición: Mantener el botón deshabilitado para evitar clics adicionales
  3. Al completar la petición: Habilitar el botón nuevamente (tanto en éxito como en error)

Ventajas de este enfoque

  • Previene envíos duplicados de formularios
  • Mejora la experiencia del usuario mostrando retroalimentación visual
  • Evita errores por operaciones concurrentes
  • Indica claramente que el sistema está procesando una solicitud

Nota: Este ejemplo utiliza la API Fetch para simular peticiones HTTP. En una aplicación real, podrías estar usando Fetch, Axios, XMLHttpRequest o cualquier otra librería para realizar peticiones.

Código JavaScript

A continuación se muestra el código JavaScript que implementa la funcionalidad:

// Función para simular una petición HTTP con delay
function simulateRequest(duration = 2000, shouldFail = false) {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            if (shouldFail) {
                reject(new Error('Error simulado en la petición'));
            } else {
                resolve({status: 'success', data: 'Petición completada correctamente'});
            }
        }, duration);
    });
}

// Función para deshabilitar un botón y mostrar spinner
function disableButton(button, text) {
    button.disabled = true;
    button.innerHTML = `<span class="loading-spinner"></span> ${text}`;
}

// Función para habilitar un botón y restaurar su texto
function enableButton(button, text) {
    button.disabled = false;
    button.innerHTML = `<span class="btn-text">${text}</span>`;
}

// Función principal para manejar peticiones
async function handleRequest(button, requestType, duration = 2000, shouldFail = false) {
    const buttonText = button.querySelector('.btn-text').textContent;
    
    // Deshabilitar el botón
    disableButton(button, `Procesando ${requestType}...`);
    
    // Actualizar estado
    updateStatus(`Procesando petición ${requestType}...`, 'status-active');
    
    try {
        // Simular petición
        const result = await simulateRequest(duration, shouldFail);
        
        // Mostrar resultado exitoso
        showResult(`✅ Petición ${requestType} completada: ${result.data}`, 'success');
        updateStatus(`Petición ${requestType} completada`, 'status-inactive');
        
    } catch (error) {
        // Mostrar error
        showResult(`❌ Error en petición ${requestType}: ${error.message}`, 'error');
        updateStatus(`Error en petición ${requestType}`, 'status-inactive');
        
    } finally {
        // Siempre habilitar el botón nuevamente (tanto en éxito como en error)
        enableButton(button, buttonText);
    }
}

// Funciones auxiliares para actualizar la UI
function updateStatus(message, cssClass) {
    const statusElement = document.getElementById('requestStatus');
    statusElement.textContent = message;
    statusElement.className = `status-indicator ${cssClass}`;
}

function showResult(message, type) {
    const resultElement = document.getElementById('result');
    resultElement.innerHTML = `<p>${message}</p>`;
    resultElement.className = `result ${type}`;
}

// Configurar event listeners cuando el DOM esté cargado
document.addEventListener('DOMContentLoaded', function() {
    const btnGet = document.getElementById('btnGet');
    const btnPost = document.getElementById('btnPost');
    const btnError = document.getElementById('btnError');
    
    btnGet.addEventListener('click', () => handleRequest(btnGet, 'GET', 1500));
    btnPost.addEventListener('click', () => handleRequest(btnPost, 'POST', 2500));
    btnError.addEventListener('click', () => handleRequest(btnError, 'ERROR', 1500, true));
});
